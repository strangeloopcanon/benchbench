def nearest_skel_point(skel, pt):

def bfs_dist(skel, start):
    H,W=skel.shape
    dist=np.full((H,W), np.inf, np.float32)
    sx,sy=start
    import heapq
    pq=[(0.0,sy,sx)]
    dist[sy,sx]=0.0
    while pq:
        d,y,x=heapq.heappop(pq)
        if d!=dist[y,x]:

        for (k,(dy,dx)) in enumerate(offs):
            ny=y+dy; nx=x+dx

    cnts,_=cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    if not cnts:
        return None
    cnt=max(cnts, key=cv2.contourArea)
    M=cv2.moments(cnt)
    if M['m00']==0:
        return None
    return (int(M['m10']/M['m00']), int(M['m01']/M['m00']))


def strip_mask(img):
    lab=cv2.cvtColor(img, cv2.COLOR_BGR2LAB).astype(np.int16)
    H,W=lab.shape[:2]
    coords=[(0,0),(0,W-1),(H-1,0),(H-1,W-1)]
    samples=[]
    for y,x in coords:
        y0=max(0,y-20); y1=min(H,y+20)
        x0=max(0,x-20); x1=min(W,x+20)
        samples.append(lab[y0:y1,x0:x1].reshape(-1,3))
    bg=np.concatenate(samples,0).mean(0)
    D=np.sqrt(((lab-bg)**2).sum(2))
    mask=(D>18).astype(np.uint8)*255
    mask=cv2.morphologyEx(mask, cv2.MORPH_OPEN, np.ones((3,3),np.uint8))
    mask=cv2.morphologyEx(mask, cv2.MORPH_CLOSE, np.ones((11,11),np.uint8))
    cnts,_=cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    if not cnts:
        return mask
    c=max(cnts, key=cv2.contourArea)
    pm=np.zeros_like(mask)
    cv2.drawContours(pm,[c],-1,255,-1)
    return pm


def find_label_boxes(img):
    # detect bright rounded rectangles (white) not including start/end rings
    hsv=cv2.cvtColor(img, cv2.COLOR_BGR2HSV)
    h,s,v=cv2.split(hsv)
    mask=((s<60) & (v>200)).astype(np.uint8)*255
    mask=cv2.morphologyEx(mask, cv2.MORPH_OPEN, np.ones((3,3),np.uint8))
    mask=cv2.morphologyEx(mask, cv2.MORPH_CLOSE, np.ones((5,5),np.uint8))
    cnts,_=cv2.findContours(mask, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    boxes=[]
    for c in cnts:
        area=cv2.contourArea(c)
        if area<200: continue
        x,y,w,h=cv2.boundingRect(c)
        ar=w/float(h)
        if w<18 or h<18: continue
        if ar<0.6 or ar>1.8: continue
        if area/(w*h) < 0.5: continue
        boxes.append((x,y,w,h,area))
    boxes=sorted(boxes, key=lambda b: -b[4])
    # remove near-duplicates by IOU
    out=[]
    for b in boxes:
        x,y,w,h,_=b
        keep=True
        for (x2,y2,w2,h2,_) in out:
            xa=max(x,x2); ya=max(y,y2)
            xb=min(x+w,x2+w2); yb=min(y+h,y2+h2)
            inter=max(0,xb-xa)*max(0,yb-ya)
            if inter/(w*h+w2*h2-inter+1e-6)>0.3:
                keep=False
                break
        if keep:
            out.append(b)
    return [(x,y,w,h) for x,y,w,h,_ in out]


def solve_item(path):
