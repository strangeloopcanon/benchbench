from PIL import Image
import numpy as np
from scipy import ndimage
from pathlib import Path

bg=np.array([246,244,238])
black=np.array([34,34,34])
colors={
    'A':np.array([225,92,87]), # red
    'B':np.array([81,150,94]), # green
    'C':np.array([74,118,201]), # blue
    'D':np.array([206,149,58]), # orange
}
red_cross=np.array([208,42,42])


def near_mask(img, color, tol=8):
    return np.all(np.abs(img.astype(int)-color.astype(int))<=tol, axis=2)

for p in sorted(Path('images').glob('item_*.ppm'))[:5]:
    img=np.array(Image.open(p).convert('RGB'))
    bgm=near_mask(img,bg,5)
    blackm=near_mask(img,black,12)
    cross=near_mask(img,red_cross,8)
    fill=~bgm & ~blackm & ~cross
    lab,n=ndimage.label(fill)
    # crosshair center from bbox midpoint
    ys,xs=np.where(cross)
    cx=int(round(xs.mean())); cy=int(round(ys.mean()))
    # choose component around ring radius 6-12
    best=None
    for r in range(2,15):
        pts=[]
        for dx in range(-r,r+1):
            for dy in (-r,r):
                pts.append((cx+dx,cy+dy))
        for dy in range(-r+1,r):
            for dx in (-r,r):
                pts.append((cx+dx,cy+dy))
        vals=[]
        for x,y in pts:
            if 0<=x<img.shape[1] and 0<=y<img.shape[0] and not cross[y,x] and fill[y,x]:
                vals.append(lab[y,x])
        vals=[v for v in vals if v>0]
        if vals:
            best=max(set(vals), key=vals.count)
            break
    # component label via badge color pixels in same component
    comp=best
    out=[]
    for letter,c in colors.items():
