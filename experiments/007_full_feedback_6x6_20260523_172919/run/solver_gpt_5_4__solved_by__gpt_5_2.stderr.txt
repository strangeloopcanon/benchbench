+    def territory_is_licensed(self, territory: str, sale_date: Date) -> bool:

+        for eff, t in self.added_territories:
+            if sale_date >= eff:
+                territories.add(t)

+    def channel_is_excluded(self, channel: str, sale_date: Date) -> bool:
+        if channel in self.excluded_channels:
+            return True
+        for eff, ch in self.excluded_channels_effective:
+            if sale_date >= eff and channel == ch:
+                return True
+        return False

+    def promo_is_allowed(self, *, sale_date: Date, format: str, channel: str) -> float | None:
+        # Returns multiplier pct if allowed at some pct, else None.
+        applicable = [r for r in self.promo_allowances if r.matches(sale_date=sale_date, format=format, channel=channel)]
+        if not applicable:
+            return None
+        # Later effective date wins; if tie, later in file wins -> stable by list order.
+        applicable.sort(key=lambda r: r.effective)
+        return applicable[-1].multiplier_pct

+    def rate_bps_for(self, *, sale_date: Date, format: str, channel: str, territory: str) -> int:
+        applicable = [r for r in self.rate_overrides if r.matches(sale_date=sale_date, format=format, channel=channel, territory=territory)]

+            return applicable[-1].rate_bps

+def recognized_base_cents_for_row(terms: ContractTerms, row: dict[str, str]) -> int:
+    channel = row["channel"]
+    units = parse_int(row["units"])
+    gross = parse_int(row["gross_cents"])
+
+    recognized = gross
+    if channel == "bundle":
+        recognized = parse_int(row["bundle_allocated_cents"])
+
+    if units == 0:
+        return 0
+
+    # discount floor check
+    fmt = row["format"]
+    list_price = parse_int(row["list_price_cents"])
+    floor_exempt = channel in terms.floor_exempt_channels
+    if not floor_exempt:
+        unit_price = abs(recognized) / abs(units)
+        floor_price = math.floor(terms.discount_floor_pct * list_price / 100.0)
+        if unit_price < (terms.discount_floor_pct * list_price / 100.0) - 1e-9:
+            recognized = (abs(units) * floor_price) * (1 if recognized >= 0 else -1)
+
+    # base multiplier
+    mult_pct = terms.base_multiplier_for(
+        sale_date=parse_date(row["sale_date"]),
+        format=fmt,
+        channel=channel,
+        territory=row["territory"],
+    )
+    if mult_pct != 100.0:
+        recognized = floor_toward_zero(recognized * (mult_pct / 100.0))
+
+    return recognized

+def row_included_and_effective_multiplier_for_promo(terms: ContractTerms, row: dict[str, str]) -> tuple[bool, float]:
+    sale_date = parse_date(row["sale_date"])
+    if not (terms.window_start <= sale_date <= terms.window_end):
+        return (False, 1.0)
+
+    territory = row["territory"]
+    if not terms.territory_is_licensed(territory, sale_date):
+        return (False, 1.0)
+
+    fmt = row["format"]
+    if fmt not in terms.licensed_formats:
+        return (False, 1.0)
+
+    channel = row["channel"]
+    if terms.channel_is_excluded(channel, sale_date):
+        return (False, 1.0)
+
+    is_promo = parse_int(row["is_promo"]) != 0
+    if not is_promo:
+        return (True, 1.0)
+
+    promo_pct = terms.promo_is_allowed(sale_date=sale_date, format=fmt, channel=channel)
+    if promo_pct is None:
+        return (False, 1.0)
+    return (True, promo_pct / 100.0)

+def compute_item(item_id: str) -> dict[str, int]:
+    base = Path("items") / item_id
+    rights = base / "rights_rider.md"
+    amendments = base / "amendments.md"
+    finance = base / "finance_memo.md"
+    sales = base / "sales_statement.csv"
+
+    terms = parse_rights_rider(rights)
+    apply_finance_memo_overrides(terms, finance)
+    parse_amendments_into(terms, amendments)
+
+    rows = parse_csv(sales)
+
+    included_units = 0
+    earned = 0
+    positive_physical_earned = 0
+
+    for row in rows:
+        included, promo_multiplier = row_included_and_effective_multiplier_for_promo(terms, row)
+        if not included:
+            continue
+
+        units = parse_int(row["units"])
+        included_units += units
+
+        recognized = recognized_base_cents_for_row(terms, row)
+        if promo_multiplier != 1.0:
+            recognized = floor_toward_zero(recognized * promo_multiplier)
+
+        rate_bps = terms.rate_bps_for(
+            sale_date=parse_date(row["sale_date"]),
+            format=row["format"],
+            channel=row["channel"],
+            territory=row["territory"],
+        )
+        row_earned = trunc_toward_zero(recognized * rate_bps / 10000.0)
+        earned += row_earned
+
+        if row["format"] in {"paperback", "hardcover"} and row_earned > 0:
+            positive_physical_earned += row_earned
+
+    reserve_withheld = math.floor(positive_physical_earned * terms.reserve_withhold_pct / 100.0)
+    recouped = min(max(earned, 0), terms.opening_advance_cents)
+    payable = max(0, earned + terms.reserve_release_cents - reserve_withheld - recouped)
+
+    return {
+        "included_units": included_units,
+        "earned_royalty_cents": earned,
+        "recouped_advance_cents": recouped,
+        "payable_cents": payable,
+    }
+

+def main() -> None:
+    items_path = Path("items_private_sample.jsonl")
+    out = []
+    for line in items_path.read_text().splitlines():

+        item = json.loads(line)
+        item_id = item["id"]
+        ans = compute_item(item_id)
+        out.append({"id": item_id, "answer": json.dumps(ans, separators=(",", ":"), sort_keys=True)})

+def recognized_base_cents_for_row(terms: ContractTerms, row: dict[str, str]) -> int:
+    channel = row["channel"]
+    units = parse_int(row["units"])
+    gross = parse_int(row["gross_cents"])
+
+    recognized = gross
+    if channel == "bundle":
+        recognized = parse_int(row["bundle_allocated_cents"])
+
+    if units == 0:
+        return 0
+
+    # discount floor check
+    fmt = row["format"]
+    list_price = parse_int(row["list_price_cents"])
+    floor_exempt = channel in terms.floor_exempt_channels
+    if not floor_exempt:
+        unit_price = abs(recognized) / abs(units)
+        floor_price = math.floor(terms.discount_floor_pct * list_price / 100.0)
+        if unit_price < (terms.discount_floor_pct * list_price / 100.0) - 1e-9:
+            recognized = (abs(units) * floor_price) * (1 if recognized >= 0 else -1)
+
+    # base multiplier
+    mult_pct = terms.base_multiplier_for(
+        sale_date=parse_date(row["sale_date"]),
+        format=fmt,
+        channel=channel,
+        territory=row["territory"],
+    )
+    if mult_pct != 100.0:
+        recognized = floor_toward_zero(recognized * (mult_pct / 100.0))
+
+    return recognized

+def row_included_and_effective_multiplier_for_promo(terms: ContractTerms, row: dict[str, str]) -> tuple[bool, float]:
+    sale_date = parse_date(row["sale_date"])
+    if not (terms.window_start <= sale_date <= terms.window_end):
+        return (False, 1.0)
+
+    territory = row["territory"]
+    if not terms.territory_is_licensed(territory, sale_date):
+        return (False, 1.0)
+
+    fmt = row["format"]
+    if fmt not in terms.licensed_formats:
+        return (False, 1.0)
+
+    channel = row["channel"]
+    if terms.channel_is_excluded(channel, sale_date):
+        return (False, 1.0)
+
+    is_promo = parse_int(row["is_promo"]) != 0
+    if not is_promo:
+        return (True, 1.0)
+
+    promo_pct = terms.promo_is_allowed(sale_date=sale_date, format=fmt, channel=channel)
+    if promo_pct is None:
+        return (False, 1.0)
+    return (True, promo_pct / 100.0)

+def compute_item(item_id: str) -> dict[str, int]:
+    base = Path("items") / item_id
+    rights = base / "rights_rider.md"
+    amendments = base / "amendments.md"
+    finance = base / "finance_memo.md"
+    sales = base / "sales_statement.csv"
+
+    terms = parse_rights_rider(rights)
+    apply_finance_memo_overrides(terms, finance)
+    parse_amendments_into(terms, amendments)
+
+    rows = parse_csv(sales)

+    included_units = 0
+    earned = 0
+    positive_physical_earned = 0
+
+    for row in rows:
+        included, promo_multiplier = row_included_and_effective_multiplier_for_promo(terms, row)
+        if not included:
+            continue
+
+        units = parse_int(row["units"])
+        included_units += units
+
+        recognized = recognized_base_cents_for_row(terms, row)
+        if promo_multiplier != 1.0:
+            recognized = floor_toward_zero(recognized * promo_multiplier)
+
+        rate_bps = terms.rate_bps_for(
+            sale_date=parse_date(row["sale_date"]),
+            format=row["format"],
+            channel=row["channel"],
+            territory=row["territory"],
+        )
+        row_earned = trunc_toward_zero(recognized * rate_bps / 10000.0)
+        earned += row_earned
+
+        if row["format"] in {"paperback", "hardcover"} and row_earned > 0:
+            positive_physical_earned += row_earned
+

+    recouped = min(max(earned, 0), terms.opening_advance_cents)
+    payable = max(0, earned + terms.reserve_release_cents - reserve_withheld - recouped)
+
+    return {
+        "included_units": included_units,
+        "earned_royalty_cents": earned,
+        "recouped_advance_cents": recouped,
+        "payable_cents": payable,
+    }

+@dataclasses.dataclass
+class ContractTerms:
+    window_start: Date
+    window_end: Date
+    licensed_territories: set[str]
+    territory_carveouts: set[str]
+    licensed_formats: set[str]
+    excluded_channels: set[str]
+    base_rates_bps: dict[str, int]
+    opening_advance_cents: int
+    reserve_release_cents: int
+    reserve_withhold_pct: float
+    discount_floor_pct: float
+    floor_exempt_channels: set[str]
+    bundle_allocation_share_pct: float
+
+    # amendments
+    excluded_channels_effective: list[tuple[Date, str]] = dataclasses.field(default_factory=list)
+    added_territories: list[tuple[Date, str]] = dataclasses.field(default_factory=list)
+    rate_overrides: list[RateRule] = dataclasses.field(default_factory=list)
+    base_multipliers: list[BaseMultiplierRule] = dataclasses.field(default_factory=list)
+    promo_allowances: list[PromoAllowanceRule] = dataclasses.field(default_factory=list)
+
+    def territory_is_licensed(self, territory: str, sale_date: Date) -> bool:
+        territories = set(self.licensed_territories)
+        for eff, t in self.added_territories:
+            if sale_date >= eff:
+                territories.add(t)
+        return (territory in territories) and (territory not in self.territory_carveouts)
+
+    def channel_is_excluded(self, channel: str, sale_date: Date) -> bool:
+        if channel in self.excluded_channels:
+            return True
+        for eff, ch in self.excluded_channels_effective:
+            if sale_date >= eff and channel == ch:
+                return True
+        return False
+
+    def promo_is_allowed(self, *, sale_date: Date, format: str, channel: str) -> float | None:
+        # Returns multiplier pct if allowed at some pct, else None.
+        applicable = [r for r in self.promo_allowances if r.matches(sale_date=sale_date, format=format, channel=channel)]
+        if not applicable:
+            return None
+        # Later effective date wins; if tie, later in file wins -> stable by list order.
+        applicable.sort(key=lambda r: r.effective)
+        return applicable[-1].multiplier_pct
+
+    def rate_bps_for(self, *, sale_date: Date, format: str, channel: str, territory: str) -> int:
+        applicable = [r for r in self.rate_overrides if r.matches(sale_date=sale_date, format=format, channel=channel, territory=territory)]
+        if applicable:
+            applicable.sort(key=lambda r: r.effective)
+            return applicable[-1].rate_bps
+        return self.base_rates_bps.get(format, 0)
+
+    def base_multiplier_for(self, *, sale_date: Date, format: str, channel: str, territory: str) -> float:
+        applicable = [r for r in self.base_multipliers if r.matches(sale_date=sale_date, format=format, channel=channel, territory=territory)]
+        if not applicable:
+            return 100.0
+        applicable.sort(key=lambda r: r.effective)
+        return applicable[-1].multiplier_pct

+def recognized_base_cents_for_row(terms: ContractTerms, row: dict[str, str]) -> int:
+    channel = row["channel"]
+    units = parse_int(row["units"])
+    gross = parse_int(row["gross_cents"])
+
+    recognized = gross
+    if channel == "bundle":
+        recognized = parse_int(row["bundle_allocated_cents"])
+
+    if units == 0:
+        return 0
+
+    # discount floor check
+    fmt = row["format"]
+    list_price = parse_int(row["list_price_cents"])
+    floor_exempt = channel in terms.floor_exempt_channels
+    if not floor_exempt:
+        unit_price = abs(recognized) / abs(units)
+        floor_price = math.floor(terms.discount_floor_pct * list_price / 100.0)
+        if unit_price < (terms.discount_floor_pct * list_price / 100.0) - 1e-9:
+            recognized = (abs(units) * floor_price) * (1 if recognized >= 0 else -1)
+
+    # base multiplier
+    mult_pct = terms.base_multiplier_for(
+        sale_date=parse_date(row["sale_date"]),
+        format=fmt,
+        channel=channel,
+        territory=row["territory"],
+    )
+    if mult_pct != 100.0:
+        recognized = floor_toward_zero(recognized * (mult_pct / 100.0))
+
+    return recognized

+def row_included_and_effective_multiplier_for_promo(terms: ContractTerms, row: dict[str, str]) -> tuple[bool, float]:
+    sale_date = parse_date(row["sale_date"])
+    if not (terms.window_start <= sale_date <= terms.window_end):
+        return (False, 1.0)
+
+    territory = row["territory"]
+    if not terms.territory_is_licensed(territory, sale_date):
+        return (False, 1.0)
+
+    fmt = row["format"]
+    if fmt not in terms.licensed_formats:
+        return (False, 1.0)
+
+    channel = row["channel"]
+    if terms.channel_is_excluded(channel, sale_date):
+        return (False, 1.0)
+
+    is_promo = parse_int(row["is_promo"]) != 0
+    if not is_promo:
+        return (True, 1.0)
+
+    promo_pct = terms.promo_is_allowed(sale_date=sale_date, format=fmt, channel=channel)
+    if promo_pct is None:
+        return (False, 1.0)
+    return (True, promo_pct / 100.0)

+def compute_item(item_id: str) -> dict[str, int]:
+    base = Path("items") / item_id
+    rights = base / "rights_rider.md"
+    amendments = base / "amendments.md"
+    finance = base / "finance_memo.md"
+    sales = base / "sales_statement.csv"
+
+    terms = parse_rights_rider(rights)
+    apply_finance_memo_overrides(terms, finance)
+    parse_amendments_into(terms, amendments)
+
+    rows = parse_csv(sales)
+
+    included_units = 0
+    earned = 0
+    positive_physical_earned = 0
+
+    for row in rows:
+        included, promo_multiplier = row_included_and_effective_multiplier_for_promo(terms, row)
+        if not included:
+            continue
+
+        units = parse_int(row["units"])
+        included_units += units
+
+        recognized = recognized_base_cents_for_row(terms, row)
+        if promo_multiplier != 1.0:
+            recognized = floor_toward_zero(recognized * promo_multiplier)
+
+        rate_bps = terms.rate_bps_for(
+            sale_date=parse_date(row["sale_date"]),
+            format=row["format"],
+            channel=row["channel"],
+            territory=row["territory"],
+        )
+        row_earned = trunc_toward_zero(recognized * rate_bps / 10000.0)
+        earned += row_earned
+
+        if row["format"] in {"paperback", "hardcover"} and row_earned > 0:
+            positive_physical_earned += row_earned
+
+    reserve_withheld = math.floor(positive_physical_earned * terms.reserve_withhold_pct / 100.0)
+    recouped = min(max(earned, 0), terms.opening_advance_cents)
+    payable = max(0, earned + terms.reserve_release_cents - reserve_withheld - recouped)
+
+    return {
+        "included_units": included_units,
+        "earned_royalty_cents": earned,
+        "recouped_advance_cents": recouped,
+        "payable_cents": payable,
+    }

+def recognized_base_cents_for_row(terms: ContractTerms, row: dict[str, str]) -> int:

+    gross = parse_int(row["gross_cents"])

+    if units == 0:

+    # discount floor check

+def compute_item(item_id: str) -> dict[str, int]:
+    base = Path("items") / item_id
+    rights = base / "rights_rider.md"
+    amendments = base / "amendments.md"

+    sales = base / "sales_statement.csv"

+
+    return {

+        "earned_royalty_cents": earned,
+        "recouped_advance_cents": recouped,

+@dataclasses.dataclass(frozen=True)
+class RateRule:
+    effective: Date
+    format: str  # 'ANY' or a format name
+    channel: str  # 'ANY' or a channel name
+    territories: set[str] | Literal["ANY"]
+    rate_bps: int
+
+    def matches(self, *, sale_date: Date, format: str, channel: str, territory: str) -> bool:
+        if sale_date < self.effective:
+            return False
+        if self.format != "ANY" and format != self.format:
+            return False
+        if self.channel != "ANY" and channel != self.channel:
+            return False
+        if self.territories != "ANY" and territory not in self.territories:

+@dataclasses.dataclass
+class ContractTerms:
+    window_start: Date
+    window_end: Date
+    licensed_territories: set[str]
+    territory_carveouts: set[str]
+    licensed_formats: set[str]
+    excluded_channels: set[str]
+    base_rates_bps: dict[str, int]
+    opening_advance_cents: int
+    reserve_release_cents: int
+    reserve_withhold_pct: float
+    discount_floor_pct: float
+    floor_exempt_channels: set[str]
+    bundle_allocation_share_pct: float
+
+    # amendments
+    excluded_channels_effective: list[tuple[Date, str]] = dataclasses.field(default_factory=list)
+    added_territories: list[tuple[Date, str]] = dataclasses.field(default_factory=list)
+    rate_overrides: list[RateRule] = dataclasses.field(default_factory=list)
+    base_multipliers: list[BaseMultiplierRule] = dataclasses.field(default_factory=list)
+    promo_allowances: list[PromoAllowanceRule] = dataclasses.field(default_factory=list)

+    def territory_is_licensed(self, territory: str, sale_date: Date) -> bool:
+        territories = set(self.licensed_territories)
+        for eff, t in self.added_territories:
+            if sale_date >= eff:
+                territories.add(t)
+        return (territory in territories) and (territory not in self.territory_carveouts)
+
+    def channel_is_excluded(self, channel: str, sale_date: Date) -> bool:
+        if channel in self.excluded_channels:
+            return True
+        for eff, ch in self.excluded_channels_effective:
+            if sale_date >= eff and channel == ch:
+                return True
+        return False
+
+    def promo_is_allowed(self, *, sale_date: Date, format: str, channel: str) -> float | None:
+        # Returns multiplier pct if allowed at some pct, else None.
+        applicable = [r for r in self.promo_allowances if r.matches(sale_date=sale_date, format=format, channel=channel)]
+        if not applicable:
+            return None
+        # Later effective date wins; if tie, later in file wins -> stable by list order.
+        applicable.sort(key=lambda r: r.effective)
+        return applicable[-1].multiplier_pct
+
+    def rate_bps_for(self, *, sale_date: Date, format: str, channel: str, territory: str) -> int:
+        applicable = [r for r in self.rate_overrides if r.matches(sale_date=sale_date, format=format, channel=channel, territory=territory)]
+        if applicable:
+            applicable.sort(key=lambda r: r.effective)
+            return applicable[-1].rate_bps
+        return self.base_rates_bps.get(format, 0)
+
+    def base_multiplier_for(self, *, sale_date: Date, format: str, channel: str, territory: str) -> float:
+        applicable = [r for r in self.base_multipliers if r.matches(sale_date=sale_date, format=format, channel=channel, territory=territory)]
+        if not applicable:
+            return 100.0
+        applicable.sort(key=lambda r: r.effective)
+        return applicable[-1].multiplier_pct

+def parse_rights_rider(path: Path) -> ContractTerms:
+    txt = path.read_text()
+    m = R_WINDOW.search(txt)
+    if not m:
+        raise ValueError(f"Missing effective window in {path}")
+    window_start = parse_date(m.group(1))
+    window_end = parse_date(m.group(2))
+
+    territories = set(_split_csvish(R_TERR.search(txt).group(1)))
+    carveouts = set(_split_csvish(R_CARVE.search(txt).group(1)))
+    formats = set(_split_csvish(R_FORMATS.search(txt).group(1)))
+    excluded_channels = set(_split_csvish(R_EXCL.search(txt).group(1)))
+
+    base_rates = {fmt: int(bps) for fmt, bps in R_RATE_LINE.findall(txt)}
+
+    opening_adv = int(R_OPEN_ADV.search(txt).group(1))
+    reserve_rel = int(R_RES_REL.search(txt).group(1))
+    reserve_w = float(R_RES_W.search(txt).group(1))
+    discount_floor = float(R_FLOOR.search(txt).group(1))
+
+    floor_exempt_txt = R_FLOOR_EX.search(txt).group(1).strip()
+    floor_exempt = set(_split_csvish(floor_exempt_txt))
+
+    bundle_share = float(R_BUNDLE_SHARE.search(txt).group(1))
+
+    return ContractTerms(
+        window_start=window_start,
+        window_end=window_end,
+        licensed_territories=territories,
+        territory_carveouts=carveouts,
+        licensed_formats=formats,

+def recognized_base_cents_for_row(terms: ContractTerms, row: dict[str, str]) -> int:
+    channel = row["channel"]
+    units = parse_int(row["units"])
+    gross = parse_int(row["gross_cents"])
+
+    recognized = gross
+    if channel == "bundle":
+        recognized = parse_int(row["bundle_allocated_cents"])
+
+    if units == 0:
+        return 0
+
+    # discount floor check
+    fmt = row["format"]
+    list_price = parse_int(row["list_price_cents"])
+    floor_exempt = channel in terms.floor_exempt_channels
+    if not floor_exempt:
+        unit_price = abs(recognized) / abs(units)
+        floor_price = math.floor(terms.discount_floor_pct * list_price / 100.0)
+        if unit_price < (terms.discount_floor_pct * list_price / 100.0) - 1e-9:
+            recognized = (abs(units) * floor_price) * (1 if recognized >= 0 else -1)
+
+    # base multiplier
+    mult_pct = terms.base_multiplier_for(
+        sale_date=parse_date(row["sale_date"]),
+        format=fmt,
+        channel=channel,
+        territory=row["territory"],
+    )
+    if mult_pct != 100.0:
+        recognized = floor_toward_zero(recognized * (mult_pct / 100.0))
+
+    return recognized

+def row_included_and_effective_multiplier_for_promo(terms: ContractTerms, row: dict[str, str]) -> tuple[bool, float]:
+    sale_date = parse_date(row["sale_date"])
+    if not (terms.window_start <= sale_date <= terms.window_end):
+        return (False, 1.0)
+
+    territory = row["territory"]
+    if not terms.territory_is_licensed(territory, sale_date):
+        return (False, 1.0)
+
+    fmt = row["format"]
+    if fmt not in terms.licensed_formats:
+        return (False, 1.0)
+
+    channel = row["channel"]
+    if terms.channel_is_excluded(channel, sale_date):
+        return (False, 1.0)
+
+    is_promo = parse_int(row["is_promo"]) != 0
+    if not is_promo:
+        return (True, 1.0)
+
+    promo_pct = terms.promo_is_allowed(sale_date=sale_date, format=fmt, channel=channel)
+    if promo_pct is None:
+        return (False, 1.0)
+    return (True, promo_pct / 100.0)

+def compute_item(item_id: str) -> dict[str, int]:
+    base = Path("items") / item_id
+    rights = base / "rights_rider.md"
+    amendments = base / "amendments.md"
+    finance = base / "finance_memo.md"
+    sales = base / "sales_statement.csv"
+
+    terms = parse_rights_rider(rights)
+    apply_finance_memo_overrides(terms, finance)
+    parse_amendments_into(terms, amendments)
+
+    rows = parse_csv(sales)
+
+    included_units = 0
+    earned = 0
+    positive_physical_earned = 0
+
+    for row in rows:
+        included, promo_multiplier = row_included_and_effective_multiplier_for_promo(terms, row)
+        if not included:
+            continue
+
+        units = parse_int(row["units"])
+        included_units += units
+
+        recognized = recognized_base_cents_for_row(terms, row)
+        if promo_multiplier != 1.0:
+            recognized = floor_toward_zero(recognized * promo_multiplier)
+
+        rate_bps = terms.rate_bps_for(
+            sale_date=parse_date(row["sale_date"]),
+            format=row["format"],
+            channel=row["channel"],
+            territory=row["territory"],
+        )
+        row_earned = trunc_toward_zero(recognized * rate_bps / 10000.0)
+        earned += row_earned
+
+        if row["format"] in {"paperback", "hardcover"} and row_earned > 0:
+            positive_physical_earned += row_earned
+
+    reserve_withheld = math.floor(positive_physical_earned * terms.reserve_withhold_pct / 100.0)
+    recouped = min(max(earned, 0), terms.opening_advance_cents)
