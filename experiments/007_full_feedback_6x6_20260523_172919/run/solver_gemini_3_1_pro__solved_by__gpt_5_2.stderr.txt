+@dataclass(frozen=True)
+class ExpenseKey:
+    day: int
+    amount_cents: int
+    description: str

+CAP_RE = re.compile(
+    r"\bcap\b.*?\bof\s+(?P<cap>\d+)\s+cents\s+per\s+sq\s*ft\s+per\s+year\b",
+    re.IGNORECASE | re.DOTALL,
+)
+CAP_RE_ALT = re.compile(
+    r"\bcapped\b.*?\bat\s+(?P<cap>\d+)\s+cents\s+per\s+(?:square\s+foot|sq\s*ft)\b.*?\bannually\b",
+    re.IGNORECASE | re.DOTALL,
+)
+
+# Examples:
+# - Regarding the 22054 cents spent on 'Snow removal 113' on Day 245: Tenant D is at fault. This is a direct charge to them.
+DIRECT_CHARGE_RE = re.compile(
+    r"Regarding\s+the\s+(?P<amount>\d+)\s+cents\s+spent\s+on\s+'(?P<desc>[^']+)'\s+on\s+Day\s+(?P<day>\d+):\s+(?P<tenant>Tenant\s+[A-Z])\b",
+    re.IGNORECASE,

+# Examples:
+# - We decided to capitalize the 'Snow removal 925' cost from Day 112. It is excluded.
+# - The 30601 cents for 'Snow removal 597' (Day 201) is actually a capital expense.
+CAPITALIZE_RE_1 = re.compile(
+    r"capitalize\s+the\s+'(?P<desc>[^']+)'\s+cost\s+from\s+Day\s+(?P<day>\d+)",
+    re.IGNORECASE,
+)
+CAPITALIZE_RE_2 = re.compile(
+    r"The\s+(?P<amount>\d+)\s+cents\s+for\s+'(?P<desc>[^']+)'\s+\(Day\s+(?P<day>\d+)\)\s+is\s+actually\s+a\s+capital\s+expense",

+# Examples:
+# - Correction: 'Security patrol 286' on Day 273 is a CapEx. Remove from OpEx.
+CORRECTION_CAPEX_RE = re.compile(
+    r"Correction:\s+'(?P<desc>[^']+)'\s+on\s+Day\s+(?P<day>\d+)\s+is\s+a\s+CapEx\b",

+def is_excluded_by_default(amount_cents: int, description: str) -> bool:
+    d = description.lower()
+    if "late fee" in d or "penalt" in d or "legal fee" in d or "attorney" in d:
+        return True
+    if amount_cents >= 500_000:

+            kw in d
+            for kw in (
+                "repair",
+                "replacement",
+                "improvement",
+                "upgrade",
+                "renovation",
+                "install",
+                "capital",

+        if key is not None:
+            direct_expenses.add(key)

+    # Capex reclassifications: sometimes only have (day, desc) or (day, amount, desc)
+    for m in CAPITALIZE_RE_1.finditer(text):
+        day = int(m.group("day"))
+        desc = m.group("desc")
+        key = find_expense_key(expenses, day=day, desc=desc)

+        amount = int(m.group("amount"))
+        desc = m.group("desc")
+        key = find_expense_key(expenses, day=day, amount=amount, desc=desc)

+def canonical_tenant(s: str) -> str:
+    s = s.strip()
+    m = re.match(r"^(Tenant\s+[A-Z])\b", s)
+    return m.group(1) if m else s

+def find_expense_key(
+    expenses: dict[ExpenseKey, int],
+    *,
+    day: int,
+    desc: str | None = None,
+    amount: int | None = None,
+) -> ExpenseKey | None:
+    desc_norm = desc.strip().lower() if desc is not None else None
+    for key in expenses.keys():
+        if key.day != day:
+            continue
+        if amount is not None and key.amount_cents != amount:
+            continue
+        if desc_norm is not None and key.description.strip().lower() != desc_norm:
+            continue
+        return key
+    return None

+def compute_sqft_days(
+    tenant: str,
+    rent_roll_rows: list[dict[str, str]],
+    expansions: list[tuple[str, str, int, int]],
+    suites_sqft: dict[str, int],
+) -> int:
+    total = 0
+    for row in rent_roll_rows:
+        if canonical_tenant(row["Tenant"]) != tenant:
+            continue
+        suite = str(int(row["Suite"]))
+        start = int(row["Start_Day"])
+        end = int(row["End_Day"])
+        days = end - start + 1
+        total += suites_sqft[suite] * days
+    for t, suite, start, end in expansions:
+        if t != tenant:
+            continue
+        days = end - start + 1
+        total += suites_sqft[suite] * days
+    return total

+def compute_item(folder: Path) -> dict[str, int]:

+        tenant = canonical_tenant(m.group("tenant"))
+        amount = int(m.group("amount"))
+        # Only count if we successfully matched to an expense key; avoids double counting malformed notes.
+        day = int(m.group("day"))
+        desc = m.group("desc")
+        key = find_expense_key(expenses, day=day, amount=amount, desc=desc)
+        if key is not None:
+            direct_charges_by_tenant.setdefault(tenant, 0)
+            direct_charges_by_tenant[tenant] += amount
+
+    base_opex_pool = 0
+    for key, amount in expenses.items():
+        if key in direct_expenses:
+            continue
+        if key in excluded_capex:
+            continue
+        if is_excluded_by_default(amount, key.description):
+            continue
+        base_opex_pool += amount
+
+    mgmt_fee = (base_opex_pool * 5) // 100
+    total_opex_pool = base_opex_pool + mgmt_fee
+
+    building_sqft_days = total_building_sqft * 365
+
+    tenants = [canonical_tenant(r["Tenant"]) for r in rent_roll_rows]
+    out: dict[str, int] = {}
+    for tenant in tenants:
+        tenant_sqft_days = compute_sqft_days(tenant, rent_roll_rows, expansions, suites_sqft)
+        base_cam = (total_opex_pool * tenant_sqft_days) // building_sqft_days
+        cap = caps.get(tenant)
+        if cap is not None:
+            max_base = (cap * tenant_sqft_days) // 365
+            if base_cam > max_base:
+                base_cam = max_base
+        final = base_cam + direct_charges_by_tenant.get(tenant, 0)
+        out[tenant] = int(final)
+    return out
